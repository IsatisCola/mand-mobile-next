# Swiper

Carousel, used to cycle through a set of pictures or cards

# Import

import { Swiper, SwiperItem } from 'mand-mobile'

Vue.createApp().component(Swiper.name, Swiper)
Vue.createApp().component(SwiperItem.name, SwiperItem)
